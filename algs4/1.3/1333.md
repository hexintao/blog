#### 1.3.32 题目：Deque。一个双向队列和栈或队列类似，但它同事支持在两端添加或删除元素。其支持的 API 有：

* `boolean isEmpty()`
* `int size()`
* `void pushLeft(Item item)`
* `void pushRight(Item item)`
* `Item popLeft`
* `Item popRight`

编写一个使用双向链表实现这份 API 的 Deque 类，以及一个使用动态数组调整实现这份 API 的 ResizinArrayDeque 类。

解答：
和之前写过的双向链表的实现基本一样，比那个还要简单些。

动态数组版：

python 的数组没有固定长度，所以这里边用一个实例变量记录最大长度进行模拟。

``` python
class ResizingArrayDeque:
    _list = list()
    # 模拟固定长度的数组
    _maxLen = 1 
    _N = 0
    def pushLeft(self, item):
        if item == None:
            print('item should not be None')
            return
        if self._N == self._maxLen:
            self.resizeList(self._N*2)
        self._list.insert(0, item)
        self._N += 1
    
    def pushRight(self, item):
        if item == None:
            print('item should not be None')
            return
        if self._N == self._maxLen:
            self.resizeList(self._N*2)
        self._list.append(item)
        self._N += 1
    
    def popLeft(self):
        if self.isEmpty():
            print('Deque is empty!')
            return
        res = self._list.pop(0)
        self._N -= 1
        if self.isEmpty():
            self._maxLen = 1
        if self._N > 0 and self._N == self._maxLen//4:
            # 数组长度调整为实际的一半
            self.resizeList(self._maxLen//2)
        return res
    
    def popRight(self):
        if self.isEmpty():
            print('Deque is empty!')
            return
        res = self._list.pop()
        self._N -= 1
        if self.isEmpty():
            self._maxLen = 1
        if self._N > 0 and self._N == self._maxLen//4:
            # 数组长度调整为实际的一半
            self.resizeList(self._maxLen//2)
        return res
        
    def isEmpty(self):
        return self._N == 0
    def size(self):
        return self._N
    
    def resizeList(self, l):
        # 模拟长度限制
        self._maxLen = l
        tempL = list()
        for i in range(0, self._N):
            tempL.append(self._list[i])
        self._list = tempL
```


双向链表版：

``` python
class Deque:
    _first = None
    _last = None
    _N = 0

    def pushLeft(self, item):
        if item == None:
            print('item should not be None')
            return
        n = DoubleNode()
        n.item = item
        if self.isEmpty():
            self._first = n
            self._last = n
        else:
            n.next = self._first
            self._first.pre = n
            self._first = n
        self._N += 1
    
    def pushRight(self, item):
        if item == None:
            print('item should not be None')
            return
        n = DoubleNode()
        n.item = item
        if self.isEmpty():
            self._first = n
            self._last = n
        else:
            n.pre = self._last
            self._last.next = n
            self._last = n
        self._N += 1
    
    def popLeft(self):
        if self.isEmpty():
            print('Deque is empty!')
            return
        n = self._first
        if self._N == 1:
            # 仅剩一个结点
            self._first = None
            self._last = None
        else:
            self._first = self._first.next
            self._first.pre = None
        self._N -= 1
        return n.item
    
    def popRight(self):
        if self.isEmpty():
            print('Deque is empty!')
            return
        n = self._last
        if self._N == 1:
            self._first = None
            self._last = None
        else:
            self._last = self._last.pre
            self._last.next = None
        self._N -= 1
        return n.item

    def isEmpty(self):
        return self._N == 0
    def size(self):
        return self._N
```
